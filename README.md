function col(a){
  const s=a.toLowerCase();
  if(/adopted|recommended\.?$|accepted|type adopted/.test(s)&&!/not\s+rec/.test(s))return"#1D9E75";
  if(/not\s+rec|rejected|failed|not adopted|adverse|revoked/.test(s))return"#E24B4A";
  if(/allotment|under test|provision made|under review|competitive test|service test|further test|filed/.test(s))return"#EF9F27";
  return"#888780";
}
function oc(a){
  const s=a.toLowerCase();
  if(/adopted|recommended\.?$|accepted|type adopted/.test(s)&&!/not\s+rec/.test(s))return"a";
  if(/not\s+rec|rejected|failed|revoked/.test(s))return"r";
  if(/allotment|under test|under review|competitive test/.test(s))return"t";
  return"o";
}

function build(){
  const cf=document.getElementById("cf").value;
  const of=document.getElementById("of").value;
  const sf=document.getElementById("sf").value.toLowerCase();
  const src=document.getElementById("src").value;

  let groups=GROUPS;
  if(cf)groups=groups.filter(g=>g.cat===cf);
  if(sf)groups=groups.filter(g=>g.name.toLowerCase().includes(sf));
  if(src){
    groups=groups.filter(g=>{
      return Object.values(g.periods).some(rlist=>rlist.some(r=>r.source===src));
    });
  }
  if(of){
    groups=groups.filter(g=>{
      return Object.values(g.periods).some(rlist=>rlist.some(r=>oc(r.action)===of));
    });
  }

  const total=groups.length;
  let acc=0,rej=0,tst=0,oth=0;
  for(const g of groups){
    const allRecs=Object.values(g.periods).flat();
    const last=allRecs[allRecs.length-1];
    const o=oc(last.action);
    if(o==="a")acc++;else if(o==="r")rej++;else if(o==="t")tst++;else oth++;
  }
  document.getElementById("stats").innerHTML=
    `<div class="stat"><div class="val">${total}</div><div class="lbl">Technologies</div></div>`+
    `<div class="stat"><div class="val" style="color:#1D9E75">${acc}</div><div class="lbl">Accepted</div></div>`+
    `<div class="stat"><div class="val" style="color:#E24B4A">${rej}</div><div class="lbl">Rejected</div></div>`+
    `<div class="stat"><div class="val" style="color:#EF9F27">${tst}</div><div class="lbl">Allotment/test</div></div>`+
    `<div class="stat"><div class="val" style="color:#888780">${oth}</div><div class="lbl">Other</div></div>`;

  const tbl=document.getElementById("tbl");
  if(!groups.length){tbl.innerHTML='<div class="none">No technologies match this filter.</div>';return;}

  // Build era column spans
  function pIdx(p){return PERIODS.indexOf(p);}
  const eraSpans=ERAS.map(e=>({label:e.label,start:pIdx(e.start),end:pIdx(e.end)}));

  let h=`<table><thead>`;
  // Era row
  h+=`<tr class="era-header"><th class="lh era-label">Era</th>`;
  for(const era of eraSpans){
    const span=era.end-era.start+1;
    h+=`<th colspan="${span}">${era.label}</th>`;
  }
  h+=`</tr>`;
  // Period header row
  h+=`<tr><th class="lh">Technology (${total})</th>`;
  for(const p of PERIODS)h+=`<th>${p}</th>`;
  h+=`</tr></thead><tbody>`;

  for(const g of groups){
    const allRecs=Object.values(g.periods).flat();
    const cnt=allRecs.length;
    h+=`<tr><td class="name-cell" title="${g.name}">${g.name}${cnt>1?`<span class="cnt">${cnt}</span>`:""}</td>`;
    for(const p of PERIODS){
      const rlist=g.periods[p];
      h+=`<td class="yc">`;
      if(rlist&&rlist.length){
        const last=rlist[rlist.length-1];
        const c=col(last.action);
        const isFin=last.source==="financial";
        const style=isFin?`background:${c};border:1.5px dashed rgba(0,0,0,.3)`:
                          `background:${c}`;
        const d=encodeURIComponent(JSON.stringify({n:g.name,a:last.action,y:p,cnt:rlist.length,src:last.source}));
        h+=`<div class="bar" style="${style}" data-d="${d}" onmouseenter="st(event,this)" onmouseleave="ht()"></div>`;
      }
      h+=`</td>`;
    }
    h+=`</tr>`;
  }
  h+=`</tbody></table>`;
  tbl.innerHTML=h;
}

function st(e,el){
  const d=JSON.parse(decodeURIComponent(el.dataset.d));
  document.getElementById("tn").textContent=d.n;
  document.getElementById("ta").textContent=d.a;
  document.getElementById("ty").textContent=d.y+(d.cnt>1?` · ${d.cnt} entries`:"");
  document.getElementById("ts").textContent=d.src==="financial"?"Source: financial data":"Source: subjects considered";
  document.getElementById("tt").style.display="block";
  mt(e);
}
function mt(e){
  const t=document.getElementById("tt");
  let x=e.clientX+14,y=e.clientY-12;
  if(x+310>window.innerWidth)x=e.clientX-320;
  if(y+100>window.innerHeight)y=e.clientY-110;
  t.style.left=x+"px";t.style.top=y+"px";
}
function ht(){document.getElementById("tt").style.display="none";}
document.addEventListener("mousemove",e=>{if(document.getElementById("tt").style.display!=="none")mt(e);});
["cf","of","src"].forEach(id=>document.getElementById(id).addEventListener("change",build));
document.getElementById("sf").addEventListener("input",build);
build();
